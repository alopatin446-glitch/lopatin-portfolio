        document.addEventListener('DOMContentLoaded', () => {
            
            // Узлы для отслеживания мыши
            const narrativeContainer = document.querySelector('.hero-narrative');
            const portraitContainer = document.querySelector('.portrait-canvas-container');
            
            let targetMouseX = 0, targetMouseY = 0;
            let currentMouseX = 0, currentMouseY = 0;
            const interpolationRatio = 0.07; // Коэффициент инерции (тяжести) движения

            // Вычисляем координаты мыши относительно центра экрана (-0.5 до 0.5)
            window.addEventListener('mousemove', (event) => {
                targetMouseX = (event.clientX / window.innerWidth) - 0.5;
                targetMouseY = (event.clientY / window.innerHeight) - 0.5;
            });

            // Единый цикл анимации (Single RAF Loop) повышенной производительности
            function updateDynamicParallax() {
                // Плавное интерполированное приближение к координатам мыши
                currentMouseX += (targetMouseX - currentMouseX) * interpolationRatio;
                currentMouseY += (targetMouseY - currentMouseY) * interpolationRatio;

                // Переводим смещение в пиксели
                const pixelShiftX = currentMouseX * 50;
                const pixelShiftY = currentMouseY * 50;

                // Вместо перезаписи transform, мы безопасно прокидываем переменные в CSS
                if (narrativeContainer) {
                    narrativeContainer.style.setProperty('--p-x', `${pixelShiftX}px`);
                    narrativeContainer.style.setProperty('--p-y', `${pixelShiftY}px`);
                }
                if (portraitContainer) {
                    portraitContainer.style.setProperty('--p-x', `${pixelShiftX}px`);
                    portraitContainer.style.setProperty('--p-y', `${pixelShiftY}px`);
                }

                requestAnimationFrame(updateDynamicParallax);
            }
            requestAnimationFrame(updateDynamicParallax);


            // ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
            // ПРОЦЕДУРНЫЙ АУДИО-ИНТЕРФЕЙС (Глубокий эмбиент-гул)
            // ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
            const audioButton = document.getElementById('audioNode');
            const audioStatusLabel = document.getElementById('audioStatusText');
            let audioContextInstance = null;
            let soundOscillator = null;
            let outputGainNode = null;
            let isSoundPlaying = false;

            function initProceduralAtmosphere() {
                audioContextInstance = new (window.AudioContext || window.webkitAudioContext)();
                
                soundOscillator = audioContextInstance.createOscillator();
                outputGainNode = audioContextInstance.createGain();
                
                soundOscillator.type = 'sine';
                // Низкочастотная фундаментальная гармоника (55 Гц) для ощущения дорогого кино-кинематографа
                soundOscillator.frequency.setValueAtTime(55, audioContextInstance.currentTime); 

                // Фильтр низких частот для бархатного, нераздражающего звука
                const filterNode = audioContextInstance.createBiquadFilter();
                filterNode.type = 'lowpass';
                filterNode.frequency.setValueAtTime(110, audioContextInstance.currentTime);

                soundOscillator.connect(filterNode);
                filterNode.connect(outputGainNode);
                outputGainNode.connect(audioContextInstance.destination);
                
                outputGainNode.gain.setValueAtTime(0, audioContextInstance.currentTime);
                soundOscillator.start();
                // Деликатное, очень медленное нарастание звука (fade-in за 4 секунды)
                outputGainNode.gain.linearRampToValueAtTime(0.05, audioContextInstance.currentTime + 4);
            }

            audioButton.addEventListener('click', () => {
                if (!isSoundPlaying) {
                    if (!audioContextInstance) {
                        initProceduralAtmosphere();
                    } else if (audioContextInstance.state === 'suspended') {
                        audioContextInstance.resume();
                        outputGainNode.gain.linearRampToValueAtTime(0.05, audioContextInstance.currentTime + 1.5);
                    }
                    audioButton.classList.add('playing');
                    audioStatusLabel.textContent = "Atmosphere On";
                    isSoundPlaying = true;
                } else {
                    if (audioContextInstance && outputGainNode) {
                        outputGainNode.gain.linearRampToValueAtTime(0, audioContextInstance.currentTime + 1);
                        setTimeout(() => {
                            if (!isSoundPlaying) audioContextInstance.suspend();
                        }, 1000);
                    }
                    audioButton.classList.remove('playing');
                    audioStatusLabel.textContent = "Atmosphere Off";
                    isSoundPlaying = false;
                }
            });
        });
    </script>
